							public function getTableValues($lowestRow, $lowestColumn, $highestRow, $highestColumn, $spreadsheet)
							{
									$worksheet = $spreadsheet->getSheet(0);
									$lowestColumnIndex = \PhpOffice\PhpSpreadsheet\Cell\Coordinate::columnIndexFromString($lowestColumn);
									$highestColumnIndex = \PhpOffice\PhpSpreadsheet\Cell\Coordinate::columnIndexFromString($highestColumn);
									$values = [];

									for ($row = $lowestRow; $row <= $highestRow; ++$row) {
											for ($col = $lowestColumnIndex; $col <= $highestColumnIndex; ++$col) {
													$value = $worksheet->getCellByColumnAndRow($col, $row)->getCalculatedValue();
													$letter = num2alpha($col - 1);
													$values[$row][$letter] = $value;
											}
									}
									return $values;
							}
